// SAAT PROSES AUTHENTICATE Pi.authenticate(['username', 'payments'], onIncompletePaymentFound) .then(function(auth) { if (!auth || !auth.user || !auth.user.uid) { // JIKA UID TIDAK ADA, LANGSUNG LEMPAR ERROR (JANGAN FALLBACK KE USERNAME) throw new Error("Otentikasi gagal: Pi SDK tidak mengembalikan UID unik."); } var uid = auth.user.uid; var username = auth.user.username; var accessToken = auth.accessToken; // Simpan ke localStorage menggunakan UID sebagai kunci utama safeSet('whs_pi_uid', uid); safeSet('whs_pi_username', username); // Hanya untuk display UI di dashboard nanti safeSet('whs_pi_token', accessToken); safeSet('whs_login_time', Date.now()); // Kirim/Upsert ke Supabase menggunakan uid return supabase.from('users').upsert({ pi_uid: uid, username: username, pi_access_token: accessToken, last_login: new Date().toISOString() }); }) .then(function(response) { if (response.error) throw response.error; // Redirect aman ke dashboard safeRedirect('dashboard.html', 500); }) .catch(function(err) { showError('Gagal Masuk', 'Aplikasi memerlukan UID resmi dari Pi Network.', 'Gagal memvalidasi akun', err.message); });
